import numpy as np

l1,l2=[],[]

n=int(input())

for i in range(n):

    l1.append(int(input()))

for i in range(n):

    l2.append(int(input()))

value1=np.array(l1)

value2=np.array(l2)

result=value1*value2

print("Product of two arrays is:",result)
